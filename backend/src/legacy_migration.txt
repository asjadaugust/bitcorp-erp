-- Migrate Projects
INSERT INTO projects (nombre, codigo_proyecto, ubicacion, estado, fecha_inicio)
SELECT 
    nombre, 
    codigo, 
    direccion, 
    CASE WHEN estado = true THEN 'active' ELSE 'completed' END,
    CURRENT_DATE
FROM tbl_g00001_unidadoperativa
ON CONFLICT (codigo_proyecto) DO NOTHING;
